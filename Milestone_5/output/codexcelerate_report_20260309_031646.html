// Function to perform bubble sort on a vector
std::vector bubble_sort(const std::vector& arr) {
    std::vector sorted_arr = arr;
    long long n = sorted_arr.size();
    for (long long i = 0; i < n; ++i) {
        for (long long j = 0; j < n - i - 1; ++j) {
            if (sorted_arr[j] > sorted_arr[j + 1]) {
                std::swap(sorted_arr[j], sorted_arr[j + 1]);
            }
        }
    }
    return sorted_arr;
}
